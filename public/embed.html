TourRadar Chat Widget

Click the blue chat button in the bottom-right corner to open the AI tour search assistant.

Embed Snippet

Copy and paste this code into any HTML page to add the TourRadar chat widget:

<!-- TourRadar Chat Widget --> <div id="tourradar-chat-widget"></div> <script> (function(){ var CHAT_URL = 'https://tourradar-app.vercel.app/chat.html'; // Floating button var btn = document.createElement('button'); btn.id = 'tr-chat-btn'; btn.setAttribute('aria-label','Open chat'); btn.innerHTML = '<svg class="chat-icon" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2">' + '<path d="M21 15a2 2 0 0 1-2 2H7l-4 4V5a2 2 0 0 1 2-2h14a2 2 0 0 1 2 2z"/></svg>' + '<svg class="close-icon" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2">' + '<line x1="18" y1="6" x2="6" y2="18"/><line x1="6" y1="6" x2="18" y2="18"/></svg>'; // Iframe var frame = document.createElement('iframe'); frame.id = 'tr-chat-frame'; frame.src = CHAT_URL; frame.allow = 'clipboard-write'; // Styles (injected so the snippet is self-contained) var css = document.createElement('style'); css.textContent = '#tr-chat-btn{position:fixed;bottom:24px;right:24px;z-index:9999;' + 'width:60px;height:60px;border-radius:50%;border:none;' + 'background:#0B82C4;color:#fff;cursor:pointer;' + 'box-shadow:0 4px 16px rgba(0,0,0,.25);display:flex;' + 'align-items:center;justify-content:center;' + 'transition:transform .15s,box-shadow .15s}' + '#tr-chat-btn:hover{transform:scale(1.08)}' + '#tr-chat-btn svg{width:28px;height:28px}' + '#tr-chat-btn .close-icon{display:none}' + '#tr-chat-btn.open .chat-icon{display:none}' + '#tr-chat-btn.open .close-icon{display:block}' + '#tr-chat-frame{position:fixed;bottom:100px;right:24px;z-index:9998;' + 'width:420px;height:620px;border:none;border-radius:16px;' + 'box-shadow:0 8px 40px rgba(0,0,0,.25);opacity:0;pointer-events:none;' + 'transform:translateY(20px) scale(.95);' + 'transition:opacity .25s,transform .25s}' + '#tr-chat-frame.open{opacity:1;pointer-events:auto;' + 'transform:translateY(0) scale(1)}' + '@media(max-width:500px){#tr-chat-frame{width:100vw;height:100vh;' + 'bottom:0;right:0;border-radius:0}' + '#tr-chat-btn{bottom:16px;right:16px}}'; document.head.appendChild(css); document.body.appendChild(frame); document.body.appendChild(btn); btn.onclick = function(){ var open = btn.classList.toggle('open'); frame.classList.toggle('open', open); btn.setAttribute('aria-label', open ? 'Close chat' : 'Open chat'); }; })(); </script>
